-- Tabelas principais do FBR CHAT

CREATE TABLE users (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name          VARCHAR(100) NOT NULL,
  email         VARCHAR(200) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  role          VARCHAR(10) DEFAULT 'user',  -- 'user' | 'admin'
  last_seen     TIMESTAMPTZ
);

CREATE TABLE agents (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name             VARCHAR(100) NOT NULL,
  slug             VARCHAR(50) UNIQUE NOT NULL,
  openclaw_config  JSONB NOT NULL,  -- model, system_prompt, temperature, max_tokens
  tts_enabled      BOOLEAN DEFAULT FALSE,
  tts_voice_id     VARCHAR(100),
  is_active        BOOLEAN DEFAULT TRUE
);

CREATE TABLE messages (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  conversation_id   UUID NOT NULL,
  conversation_type VARCHAR(5) CHECK (conversation_type IN ('group', 'pvt')),
  sender_type       VARCHAR(10) CHECK (sender_type IN ('user', 'agent')),
  sender_id         UUID NOT NULL,
  content           TEXT,         -- texto ou transcrição STT
  media_url         TEXT,         -- URL MinIO para áudio/vídeo/imagem
  media_type        VARCHAR(20),  -- 'audio' | 'video' | 'image'
  tts_audio_url     TEXT,         -- URL do áudio TTS gerado (agentes)
  created_at        TIMESTAMPTZ DEFAULT NOW()
);

Reconexão com exponential backoff

MIME forjado rejeitado com 415
500 mensagens sem frame drops
Scroll lock quando usuário rolou

Fluxo WebSocket — Mensagem com Agente em GrupoFLOW
1. Usuário emite send_message { room_type: 'group', room_id, content }
2. Backend persiste a mensagem no PostgreSQL (ANTES de entregar)
3. Backend emite message_received para todos os membros conectados
4. Backend verifica se agente é membro do grupo
5. Se sim:
   a. Emite agent_typing → indicador "..." aparece para todos
   b. Aguarda debounce de 3s (agrupa mensagens consecutivas do user)
   c. Busca últimas 20 mensagens do grupo como contexto
   d. Chama callOpenClaw(agent, messages) com timeout de 10s
   e. Se timeout: emite message_received com erro legível
   f. Se sucesso: persiste e emite message_received sender_type='agent'
   g. Emite agent_done_typing
6. Se agente tem tts_enabled:
   a. Chama ElevenLabs TTS de forma assíncrona
   b. Salva MP3 no MinIO
   c. Emite message_updated { message_id, tts_audio_url }

ContextRouter.build_pvt_context() — Algoritmo CompletoPYTHON
def build_pvt_context(user_id, agent_id, message):
    # 1. Tentar identificar o tema da mensagem por palavras-chave
    tema = identify_tema(message)  # retorna None se não identificado

    # 2. Ler arquivos de memória do usuário
    memory  = read_file(f"memoria/usuarios/{user_id}/MEMORY.md") or ""
    history = read_file(f"memoria/usuarios/{user_id}/HISTORY.md") or ""

    # 3. Filtrar por tema (busca keywords) — máximo 2000 tokens
    mem_filtered  = filter_by_tema(memory, tema) if tema else memory[:2000]
    hist_filtered = filter_history(history, agent_id, tema)

    if not tema:
        # Tema não identificado → injetar instrução de clarificação
        return """INSTRUÇÃO: Antes de responder, pergunte:
        'Para que eu contextualize corretamente, sobre qual assunto você está falando?'"""

    return f"""
    === Contexto sobre o usuário (tema: {tema}) ===
    O que este usuário expressou nos grupos:
    {mem_filtered or "Sem contexto disponível ainda."}

    Histórico de conversas anteriores sobre este tema:
    {hist_filtered or "Esta é a primeira conversa sobre este tema."}

    INSTRUÇÃO: Use o contexto acima. Não mencione que tem acesso a ele.
    """
🔒
Regra Inviolável de Privacidade
MEMORY.md nunca é injetado em contexto de grupo. HISTORY.md (PVT privado) nunca é referenciado em grupo. O agente filtra apenas o que o próprio usuário disse coletivamente. Contexto de PVT de outros usuários nunca é cruzado.
